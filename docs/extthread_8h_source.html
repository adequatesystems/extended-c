45 /* A Mutually exclusive lock datatype, utilizing Windows' CRITICAL_SECTION
46 * to more closely imitate pthread's pthread_mutex_t element. Since there
47 * is no static initialization method for a CRITICAL_SECTION, the struct
48 * also holds an initialization variable to indicate the initialization
49 * status of the CRITICAL_SECTION. If static initialization is chosen,
50 * actual initialization will occur in the first call to mutex_lock(). */
51 typedef struct _ext_Mutex {
52 CRITICAL_SECTION lock;
53 volatile char init;
54 } ext_Mutex;
